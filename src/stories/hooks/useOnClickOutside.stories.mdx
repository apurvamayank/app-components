# ReactJS Hooks: useOnClickOutside

````jsx harmony
import React, {useRef, useState} from 'react';
import {useOnClickOutside} from '@datorama/app-components';

const Comp = () => {
  const [open, setOpen] = useState(false);
  const ref = useRef(null);
  const handleClickOut = () => setOpen(false);
  
  // (ref, handler, isActive) 
  useOnClickOutside(ref, handleClickOut, open);
  
  return <Popup ref={ref} />;
};
````
